H5Fclose.c

After creating an HDF5 file with H5Fcreate(), we close it with H5Fclose().

#include "hdf5.h"
int main()
{
hid_t file;
if ((file = H5Fcreate("foo.h5", H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT)) < 0)
return -1;
/* Do something good with this file. */
if(H5Fclose(file) < 0)
return -2;
return 0;
}
